/** * ระบบ Backend สำหรับจัดการใบเสนอราคา * จัดการโดย: กิต โซลูชั่น จำกัด */ const FOLDER_ID = "1OZDzFqsQ_55Xipuyi5uxxpmVqGBylML1"; // โฟลเดอร์หลักใน Google Drive สำหรับเก็บใบเสนอราคา const SHEET_NAME = "History"; const SETTINGS_SHEET_NAME = "Settings"; /** * รับค่าจาก Frontend (index.html) */ function doPost(e) { try { const data = JSON.parse(e.postData.contents); // 1. สร้างใบเสนอราคาและบันทึก PDF if (data.action === 'createQuotation') { const pdfUrl = generatePDFAndSave(data); saveToSheet(data, pdfUrl); return ContentService.createTextOutput(JSON.stringify({ status: 'success', url: pdfUrl })).setMimeType(ContentService.MimeType.JSON); } // 2. บันทึกการตั้งค่าบริษัทลง Sheet else if (data.action === 'saveSettings') { saveCompanySettings(data.settings); return ContentService.createTextOutput(JSON.stringify({ status: 'success' })).setMimeType(ContentService.MimeType.JSON); } // 3. ดึงการตั้งค่าบริษัทจาก Sheet else if (data.action === 'getSettings') { const settings = getCompanySettings(); return ContentService.createTextOutput(JSON.stringify({ status: 'success', settings: settings })).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: 'ไม่พบคำสั่งที่ต้องการ' })).setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } /** * บันทึกการตั้งค่าลง Sheet (Settings) */ function saveCompanySettings(settings) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SETTINGS_SHEET_NAME) || ss.insertSheet(SETTINGS_SHEET_NAME); sheet.clear(); const dataToSave = [ ["key", "value"], ["name", settings.name || ""], ["phone", settings.phone || ""], ["email", settings.email || ""], ["lineId", settings.lineId || ""], ["defaultRemark", settings.defaultRemark || ""], ["defaultSignerName", settings.defaultSignerName || ""] ]; sheet.getRange(1, 1, dataToSave.length, 2).setValues(dataToSave); } /** * ดึงการตั้งค่าจาก Sheet (Settings) */ function getCompanySettings() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SETTINGS_SHEET_NAME); if (!sheet) return null; const data = sheet.getDataRange().getValues(); let settings = {}; for (let i = 1; i < data.length; i++) { if (data[i][0]) settings[data[i][0]] = String(data[i][1]); } return settings; } /** * สร้าง PDF และบันทึกลงโฟลเดอร์ลูกค้า */ function generatePDFAndSave(data) { const rootFolder = DriveApp.getFolderById(FOLDER_ID); // ค้นหาหรือสร้างโฟลเดอร์ย่อยตามชื่อลูกค้า let customerFolders = rootFolder.searchFolders(`title = '${data.customer}'`); let customerFolder = customerFolders.hasNext() ? customerFolders.next() : rootFolder.createFolder(data.customer); // สร้างโครงสร้าง HTML สำหรับ PDF let html = `

ใบเสนอราคา (Quotation)

ผู้เสนอราคา:
${data.company.name}
โทร: ${data.company.phone}
${data.company.email ? 'อีเมล: ' + data.company.email : ''} ${data.company.lineId ? '
LINE: ' + data.company.lineId : ''} 		วันที่: ${new Date().toLocaleDateString('th-TH')}
ลูกค้า: ${data.customer}
${data.projectName ? 'โครงการ: ' + data.projectName : ''}
`; data.items.forEach((item, index) => { const qtyText = item.isLumpSum ? '-' : item.qty; const priceText = item.isLumpSum ? '-' : item.price.toLocaleString(); const itemTotal = item.isLumpSum ? item.price : (item.qty * item.price); html += ` `; }); html += `
ลำดับ รายการสินค้า/บริการ จำนวน ราคา/หน่วย รวมเป็นเงิน
${index + 1} ${item.name} ${item.isLumpSum ? '(ราคาเหมา)' : ''} ${qtyText} ${priceText} ${itemTotal.toLocaleString()}
หมายเหตุ / เงื่อนไข:
${data.remark || '-'}

ยอดรวมสุทธิ: ${data.total.toLocaleString()} บาท

(ภาษีมูลค่าเพิ่ม 0%)
${data.company.defaultSignerName || ''}

ผู้เสนอราคา

ผู้อนุมัติ / ลูกค้า
`; // แปลง HTML เป็น PDF และบันทึก const blob = HtmlService.createHtmlOutput(html).getAs('application/pdf').setName(`ใบเสนอราคา_${data.customer}_${new Date().getTime()}.pdf`); const file = customerFolder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return file.getUrl(); } /** * บันทึกประวัติการสร้างลง Sheet (History) */ function saveToSheet(data, pdfUrl) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME); if (sheet.getLastRow() === 0) { sheet.appendRow(["วันที่", "ลูกค้า", "โครงการ", "ยอดรวม", "ลิงก์ PDF"]); } sheet.appendRow([new Date(), data.customer, data.projectName || '-', data.total, pdfUrl]); } function doGet(e) { return ContentService.createTextOutput("ระบบทำงานปกติ").setMimeType(ContentService.MimeType.TEXT); }
